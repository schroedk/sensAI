92 # normalise data

94 self.outputScaler = normalisation.VectorDataScaler(outputs, self.normalisationMode)

96 normOutputs = self.outputScaler.getNormalisedArray(outputs) 

97 

98 # split data into training and validation set 

99 trainSplit = int(normInputs.shape[0] * (1-self.validationFraction)) 

100 trainInputs = normInputs[:trainSplit] 

101 trainOutputs = normOutputs[:trainSplit] 

102 valInputs = normInputs[trainSplit:] 

103 valOutputs = normOutputs[trainSplit:] 

104 

105 # create and fit model 

106 TensorFlowSession.setKerasSession() 

107 model = self._createModel(inputs.shape[1], outputs.shape[1]) 

108 model.compile(optimizer=self.optimiser, loss=self.loss, metrics=self.metrics) 

109 tempFileHandle, tempFilePath = tempfile.mkstemp(".keras.model")

112 checkpointCallback = tf.keras.callbacks.ModelCheckpoint(tempFilePath, monitor='val_loss', save_best_only=True, save_weights_only=True)
