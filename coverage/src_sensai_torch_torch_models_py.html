71class LSTNetworkVectorClassificationModel(TorchVectorClassificationModel): 

72 """ 

73 Classification model for time series data using the LSTNetwork architecture. 

74 

75 Since the model takes a time series as input, it requires that input data frames to use special naming of columns 

76 such that the data can be interpreted correctly: 

77 Each column name must start with an N-digit prefix indicating the time slice the data pertains to (for any fixed N); 

78 the following suffix shall indicate the name of the actual feature. 

79 For each N-digit prefix, we must have the same set of suffixes in the list of columns, i.e. we must have the same 

80 features for each time slice in the input time series. 

81 """ 

82 def __init__(self, numInputTimeSlices, inputDimPerTimeSlice, numClasses: Optional[int] = None, 

83 numConvolutions: int = 100, numCnnTimeSlices: int = 6, hidRNN: int = 100, skip: int = 0, hidSkip: int = 5, 

84 hwWindow: int = 0, hwCombine: str = "plus", dropout=0.2, outputActivation="sigmoid", 

85 nnOptimiserParams: dict = None): 

86 """ 

87 :param numInputTimeSlices: the number of input time slices 

88 :param inputDimPerTimeSlice: the dimension of the input data per time slice 

89 :param numClasses: the number of classes considered by this classification problem; if None, determine from data 

90 :param numCnnTimeSlices: the number of time slices considered by each convolution (i.e. it is one of the dimensions of the matrix used for 

91 convolutions, the other dimension being inputDimPerTimeSlice), a.k.a. "Ck" 

92 :param numConvolutions: the number of separate convolutions to apply, i.e. the number of independent convolution matrices, a.k.a "hidC"; 

93 if it is 0, then the entire complex processing path is not applied. 

94 :param hidRNN: the number of hidden output dimensions for the RNN stage 

95 :param skip: the number of time slices to skip for the skip-RNN. If it is 0, then the skip-RNN is not used. 

96 :param hidSkip: the number of output dimensions of each of the skip parallel RNNs 

97 :param hwWindow: the number of time slices from the end of the input time series to consider as input for the highway component. 

98 If it is 0, the highway component is not used. 

99 :param hwCombine: {"plus", "product", "bilinear"} the function with which the highway component's output is combined with the complex path's output 

100 :param dropout: the dropout probability to use during training (dropouts are applied after every major step in the evaluation path) 

101 :param outputActivation: the output activation function 

102 :param nnOptimiserParams: parameters of NNOptimiser to use for training 

103 """ 

104 if nnOptimiserParams is None: 

105 nnOptimiserParams = {} 

106 self.cuda = nnOptimiserParams.get("cuda", False) 

107 self.numClasses = numClasses 

108 lstnetArgs = dict(numInputTimeSlices=numInputTimeSlices, inputDimPerTimeSlice=inputDimPerTimeSlice, numOutputTimeSlices=1, 

109 outputDimPerTimeSlice=numClasses, numConvolutions=numConvolutions, numCnnTimeSlices=numCnnTimeSlices, hidRNN=hidRNN, 

110 hwWindow=hwWindow, hwCombine=hwCombine, dropout=dropout, outputActivation=outputActivation, 

111 skip=skip, hidSkip=hidSkip, isClassification=True) 

112 super().__init__(self._LSTNetworkModel, modelArgs=[self.cuda], modelKwArgs=lstnetArgs, nnOptimiserParams=nnOptimiserParams)

151 def _computeInputs(self, x: pd.DataFrame, y=None, fit=False) -> pd.DataFrame: 

152 x = super()._computeInputs(x, Y=y, fit=fit) 

153 

154 # sort input data frame columns by name 

155 x = x[sorted(x.columns)] 

156 

157 # check input column name format and consistency 

158 colNameRegex = re.compile(r"(\d+).+") 

159 colsByTimeSlice = collections.defaultdict(list) 

160 numDigits = None 

161 for colName in x.columns: 

162 match = colNameRegex.fullmatch(colName) 

163 if not match: 

164 raise ValueError(f"Column name '{colName}' does not match the required format (N-digit prefix indicating the time slice order followed by feature name; for any fixed N); columns={list(x.columns)}") 

165 timeSlice = match.group(1) 

166 if numDigits is None: 

167 numDigits = len(timeSlice) 

168 elif numDigits != len(timeSlice): 

169 raise ValueError(f"Inconsistent number of digits in column names: Got {numDigits} leading digits for one feature and {len(timeSlice)} for another; columns={list(x.columns)}") 

170 colsByTimeSlice[timeSlice].append(colName[numDigits:]) 

171 referenceCols = None 

172 for timeSlice, cols in colsByTimeSlice.items(): 

173 if referenceCols is None: 

174 referenceCols = cols 

175 elif referenceCols != cols: 

176 raise ValueError(f"Inconsistent features across time slices: Got suffixes {cols} for one time slice and {referenceCols} for another; columns={list(x.columns)}") 

177 

178 return x
