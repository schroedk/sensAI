59 def _loadModel(self, modelFile): 

60 try: 

61 self.module = torch.load(modelFile) 

62 self._gpu = self._getGPUFromModelParameterDevice() 

63 except: 

64 if self._isCudaEnabled(): 

65 if torch.cuda.device_count() > 0: 

66 newDevice = "cuda:0" 

67 else: 

68 newDevice = "cpu" 

69 self.log.warning(f"Loading of CUDA model failed, trying to map model to device {newDevice}...") 

70 if type(modelFile) != str: 

71 modelFile.seek(0) 

72 try: 

73 self.module = torch.load(modelFile, map_location=newDevice) 

74 except: 

75 self.log.warning(f"Failure to map model to device {newDevice}, trying CPU...") 

76 if newDevice != "cpu": 

77 newDevice = "cpu" 

78 self.module = torch.load(modelFile, map_location=newDevice) 

79 if newDevice == "cpu": 

80 self._setCudaEnabled(False) 

81 self._gpu = None 

82 else: 

83 self._gpu = 0 

84 self.log.info(f"Model successfully loaded to {newDevice}") 

85 else: 

86 raise

107 def apply(self, X: Union[torch.Tensor, np.ndarray, TorchDataSet], asNumpy=True, createBatch=False, mcDropoutSamples=None, mcDropoutProbability=None, scaleOutput=False, 

108 scaleInput=False) -> Union[torch.Tensor, np.ndarray, Tuple]: 

109 """ 

110 Applies the model to the given input tensor and returns the result (normalized) 

111 

112 :param X: the input tensor (either a batch or, if createBatch=True, a single data point) or data set. 

113 If it is a data set, a single tensor will be extracted from it, so the data set must not be too large to be processed at once. 

114 :param asNumpy: flag indicating whether to convert the result to a numpy.array (if False, return tensor) 

115 :param createBatch: whether to add an additional tensor dimension for a batch containing just one data point 

116 :param mcDropoutSamples: if not None, apply MC-Dropout-based inference with the respective number of samples; if None, apply regular inference 

117 :param mcDropoutProbability: the probability with which to apply dropouts in MC-Dropout-based inference; if None, use model's default 

118 :param scaleOutput: whether to scale the output that is produced by the underlying model (using this instance's output scaler) 

119 :param scaleInput: whether to scale the input (using this instance's input scaler) before applying the underlying model 

120 

121 :return: an output tensor or, if MC-Dropout is applied, a pair (y, sd) where y the mean output tensor and sd is a tensor of the same dimension 

122 containing standard deviations 

123 """ 

124 def extract(z): 

125 if scaleOutput: 

126 z = self.scaledOutput(z) 

127 if self._isCudaEnabled(): 

128 z = z.cpu() 

129 z = z.detach() 

130 if asNumpy: 

131 z = z.numpy() 

132 return z 

133 

134 model = self.getTorchModule() 

135 model.eval() 

136 

137 if isinstance(X, TorchDataSet): 

138 X = next(X.iterBatches(X.size(), inputOnly=True, shuffle=False)) 

139 elif isinstance(X, np.ndarray): 

140 X = toFloatArray(X) 

141 X = torch.from_numpy(X).float() 

142 

143 if self._isCudaEnabled(): 

144 torch.cuda.set_device(self._gpu) 

145 X = X.cuda() 

146 if scaleInput: 

147 X = self.inputScaler.normalise(X) 

148 if createBatch: 

149 X = X.view(1, *X.size()) 

150 

151 maxValue = X.max().item() 

152 if maxValue > 2: 

153 log.warning("Received input which is likely to not be correctly normalised: maximum value in input tensor is %f" % maxValue) 

154 

155 if mcDropoutSamples is None: 

156 y = model(X) 

157 return extract(y) 

158 else: 

159 y, stddev = model.inferMCDropout(X, mcDropoutSamples, p=mcDropoutProbability) 

160 return extract(y), extract(stddev)

172 return self.apply(X, scaleOutput=True, scaleInput=True, **kwargs)
