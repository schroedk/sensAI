79 def step(self, lossBackward: Callable): 

80 """ 

81 

82 :param lossBackward: callable, performs backward step and returns loss 

83 :return: 

84 """ 

85 if self.use_shrinkage: 

86 def closureWithShrinkage(): 

87 loss = lossBackward() 

88 

89 # Compute gradients norm. 

90 grad_norm = 0 

91 for param in self.params: 

92 grad_norm += math.pow(param.grad.data.norm(), 2) 

93 

94 grad_norm = math.sqrt(grad_norm) 

95 if grad_norm > 0: 

96 shrinkage = self.max_grad_norm / grad_norm 

97 else: 

98 shrinkage = 1. 

99 

100 for param in self.params: 

101 if shrinkage < 1: 

102 param.grad.data.mul_(shrinkage) 

103 

104 return loss 

105 

106 closure = closureWithShrinkage 

107 else: 

108 closure = lossBackward 

109 

110 loss = self.optimizer.step(closure) 

111 return loss

230 class ValidationLossEvaluator(NNLossEvaluator.ValidationLossEvaluator): 

231 def __init__(self, cuda: bool): 

232 self.total_loss_l1 = None 

233 self.total_loss_l2 = None 

234 self.outputDims = None 

235 self.allTrueOutputs = None 

236 self.evaluateL1 = nn.L1Loss(reduction='sum') 

237 self.evaluateL2 = nn.MSELoss(reduction='sum') 

238 if cuda: 

239 self.evaluateL1 = self.evaluateL1.cuda() 

240 self.evaluateL2 = self.evaluateL2.cuda() 

241 

242 def startValidationCollection(self, groundTruthShape): 

243 if len(groundTruthShape) != 1: 

244 raise ValueError("Outputs that are not vectors are currently unsupported") 

245 self.outputDims = groundTruthShape[-1] 

246 self.total_loss_l1 = np.zeros(self.outputDims) 

247 self.total_loss_l2 = np.zeros(self.outputDims) 

248 self.allTrueOutputs = None 

249 

250 def processValidationResultBatch(self, output, groundTruth): 

251 # obtain series of outputs per output dimension: (batch_size, output_size) -> (output_size, batch_size) 

252 predictedOutput = output.permute(1, 0) 

253 trueOutput = groundTruth.permute(1, 0) 

254 

255 if self.allTrueOutputs is None: 

256 self.allTrueOutputs = trueOutput 

257 else: 

258 self.allTrueOutputs = torch.cat((self.allTrueOutputs, trueOutput), dim=1) 

259 

260 for i in range(self.outputDims): 

261 self.total_loss_l1[i] += self.evaluateL1(predictedOutput[i], trueOutput[i]).item() 

262 self.total_loss_l2[i] += self.evaluateL2(predictedOutput[i], trueOutput[i]).item() 

263 

264 def endValidationCollection(self): 

265 outputDims = self.outputDims 

266 rae = np.zeros(outputDims) 

267 rrse = np.zeros(outputDims) 

268 mae = np.zeros(outputDims) 

269 mse = np.zeros(outputDims) 

270 

271 for i in range(outputDims): 

272 mean = torch.mean(self.allTrueOutputs[i]) 

273 refModelErrors = self.allTrueOutputs[i] - mean 

274 refModelSumAbsErrors = torch.sum(torch.abs(refModelErrors)).item() 

275 refModelSumSquaredErrors = torch.sum(refModelErrors * refModelErrors).item() 

276 numSamples = refModelErrors.size(0) 

277 

278 mae[i] = self.total_loss_l1[i] / numSamples 

279 mse[i] = self.total_loss_l2[i] / numSamples 

280 rae[i] = self.total_loss_l1[i] / refModelSumAbsErrors if refModelSumAbsErrors != 0 else np.inf 

281 rrse[i] = np.sqrt(mse[i]) / np.sqrt( 

282 refModelSumSquaredErrors / numSamples) if refModelSumSquaredErrors != 0 else np.inf 

283 

284 metrics = OrderedDict([("RRSE", np.mean(rrse)), ("RAE", np.mean(rae)), ("MSE", np.mean(mse)), ("MAE", np.mean(mae))]) 

285 return metrics 

286 

287 def getValidationMetricName(self): 

288 if self.lossFn is self.LossFunction.L1LOSS or self.lossFn is self.LossFunction.SMOOTHL1LOSS: 

289 return "MAE" 

290 elif self.lossFn is self.LossFunction.L2LOSS or self.lossFn is self.LossFunction.MSELOSS: 

291 return "MSE" 

292 else: 

293 raise AssertionError(f"No selection criterion defined for loss function {self.lossFn}")

561 def _train(self, dataSets: Sequence[TorchDataSet], model: nn.Module, criterion: nn.modules.loss._Loss, 

562 optim: _Optimiser, batch_size: int, cuda: bool, outputScalers: Sequence[TensorScaler]): 

563 """Performs one training epoch""" 

564 model.train() 

565 total_loss = 0 

566 n_samples = 0 

567 numOutputsPerDataPoint = None 

568 for dataSet, outputScaler in zip(dataSets, outputScalers): 

569 for X, Y in dataSet.iterBatches(batch_size, shuffle=True): 

570 if numOutputsPerDataPoint is None: 

571 outputShape = Y.shape[1:] 

572 numOutputsPerDataPoint = functools.reduce(lambda x, y: x * y, outputShape, 1) 

573 

574 def closure(): 

575 model.zero_grad() 

576 output, groundTruth = self._applyModel(model, X, Y, outputScaler) 

577 loss = criterion(output, groundTruth) 

578 loss.backward() 

579 return loss 

580 

581 loss = optim.step(closure) 

582 total_loss += loss.item() 

583 numDataPointsInBatch = Y.size(0) 

584 n_samples += numDataPointsInBatch * numOutputsPerDataPoint 

585 return total_loss / n_samples

587 def _evaluate(self, dataSet: TorchDataSet, model: nn.Module, outputScaler: TensorScaler): 

588 """Evaluates the model on the given data set (a validation set)""" 

589 model.eval() 

590 

591 groundTruthShape = None 

592 for X, Y in dataSet.iterBatches(self.batchSize, shuffle=False): 

593 if groundTruthShape is None: 

594 groundTruthShape = Y.shape[1:] # the shape of the output of a single model application 

595 self.lossEvaluatorState.startValidationCollection(groundTruthShape) 

596 with torch.no_grad(): 

597 output, groundTruth = self._applyModel(model, X, Y, outputScaler) 

598 self.lossEvaluatorState.processValidationResultBatch(output, groundTruth) 

599 

600 return self.lossEvaluatorState.endValidationCollection()
